CREATE TABLE `ks_user` (  `id` int(11) NOT NULL AUTO_INCREMENT,  `realname` varchar(15) NOT NULL COMMENT '真实姓名',  `phone` varchar(15) NOT NULL COMMENT '电话',  `idcard` varchar(18) NOT NULL COMMENT '身份证号',  `email` varchar(50) NOT NULL COMMENT '邮箱',  `password` varchar(32) NOT NULL COMMENT '密码',  `formal` tinyint(4) NOT NULL DEFAULT '-1' COMMENT '-1：非正式   1：正式',  `is_del` tinyint(4) NOT NULL DEFAULT '1' COMMENT '-1：删除   1：正常',  `time_length` int(11) NOT NULL COMMENT '剩余时长',  PRIMARY KEY (`id`)) ENGINE=MyISAM DEFAULT CHARSET=utf8 COMMENT='用户表'
